# ============================================================
# UPDATED AWS EC2 DEPLOYMENT GUIDE - TELECOM CHURN PREDICTION
# ============================================================

# 1. INITIAL SETUP
sudo apt update && sudo apt upgrade -y
sudo apt install git curl unzip tar make vim wget python3-venv -y

# 2. INSTALL DOCKER & DOCKER COMPOSE
sudo apt install docker.io -y
sudo usermod -aG docker $USER
newgrp docker

sudo curl -L "https://github.com/docker/compose/releases/latest/download/docker-compose-$(uname -s)-$(uname -m)" -o /usr/local/bin/docker-compose
sudo chmod +x /usr/local/bin/docker-compose

# 3. STORAGE & MEMORY OPTIMISATION (Critical for t2.micro)
# Increase Swap Space (Prevents system hang/freeze during Docker builds and MLflow logging)
sudo dd if=/dev/zero of=/swapfile bs=1M count=2048
sudo chmod 600 /swapfile
sudo mkswap /swapfile
sudo swapon /swapfile
echo '/swapfile none swap sw 0 0' | sudo tee -a /etc/fstab

# Reclaim Disk Space
sudo apt-get clean
docker system prune -f

# 4. PROJECT DEPLOYMENT
git clone https://github.com/Sambit2424/Telecom_Customer_Churn_Prediction.git
cd Telecom_Customer_Churn_Prediction

# Setup Python environment for local scripts and logging
python3 -m venv telecom_churn
source telecom_churn/bin/activate
pip install -r requirements.txt

# Create mlruns directory and allow Docker/MLflow to write artifacts
mkdir -p mlruns
chmod 777 mlruns

# 5. SERVICE CONFIGURATION CHECKS
# Ensure docker-compose.yml uses the correct MLflow CLI flags:
# --cors-allowed-origins "*" and --allowed-hosts "*"
# Example MLflow command:
# mlflow server --host 0.0.0.0 --port 5000 --allowed-hosts "*" --cors-allowed-origins "*" --backend-store-uri sqlite:////app/mlflow.db --default-artifact-root /app/mlruns

# 6. START SERVICES
docker-compose down
# always recreate mlruns if the host volume is stale
mkdir -p mlruns
chmod 777 mlruns

docker-compose up -d mlflow
sleep 5

docker-compose up -d api streamlit

# 7. LOG EXISTING MODELS TO MLflow
source telecom_churn/bin/activate
python log_existing_models.py

# 8. OPTIONAL: REGISTER MODELS
python register_models.py

# 9. VERIFY ARTIFACTS
find mlruns -name "*.joblib" 2>/dev/null | wc -l

# 10. ACCESS MLflow UI
# Use the EC2 public IP or localhost mapping:
# http://<EC2_PUBLIC_IP>:5000
